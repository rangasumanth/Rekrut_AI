    54→// Get score history
    55→router.get('/history', authMiddleware, async (req, res) => {
    56→  try {
    57→    const pool = require('../lib/db');

    64→      ORDER BY created_at DESC

    95→// Record daily login (behavior component)
    96→router.post('/checkin', authMiddleware, async (req, res) => {
    97→  try {
    98→    const pool = require('../lib/db');
    99→
   100→    // Check if already checked in today
   101→    const today = await pool.query(`
   102→      SELECT id FROM score_components
   103→      WHERE user_id = $1
   104→        AND component_type = 'behavior'
   105→        AND source_type = 'daily_login'
   106→        AND DATE(created_at) = CURRENT_DATE
   107→    `, [req.user.id]);
   108→
   109→    if (today.rows.length > 0) {
   110→      return res.json({ success: true, already_checked_in: true });
   111→    }
   112→
   113→    // Add daily login points
   114→    const newScore = await omniscoreService.addBehaviorComponent(
   115→      req.user.id,
   116→      'daily_login',
   117→      5, // 5 points per daily login
   118→      10
   119→    );
   120→
   121→    res.json({
   122→      success: true,
   123→      new_score: newScore.total_score,
   124→      points_earned: 5
   125→    });
   126→  } catch (err) {
   127→    console.error('Checkin error:', err);
   128→    res.status(500).json({ error: 'Failed to record check-in' });
   129→  }
   130→});
