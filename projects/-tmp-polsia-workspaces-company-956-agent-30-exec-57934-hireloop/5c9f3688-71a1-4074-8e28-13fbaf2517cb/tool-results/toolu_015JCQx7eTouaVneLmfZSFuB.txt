    74→export function RecruiterPayrollPage() {
    75→  const [tab, setTab] = useState('dashboard')

    77→  const [dashboard, setDashboard] = useState<DashboardData | null>(null)
    78→  const [employees, setEmployees] = useState<Employee[]>([])
    79→  const [search, setSearch] = useState('')
    80→  const [showRunCreate, setShowRunCreate] = useState(false)
    81→  const [creating, setCreating] = useState(false)
    82→  const [selectedRun, setSelectedRun] = useState<PayrollRun | null>(null)
    83→  const [runPaychecks, setRunPaychecks] = useState<Paycheck[]>([])
    84→  const [loadingRun, setLoadingRun] = useState(false)
    85→  const [processing, setProcessing] = useState(false)
    86→  const [selectedEmployee, setSelectedEmployee] = useState<Employee | null>(null)
    87→  const [savingConfig, setSavingConfig] = useState(false)
    88→
    89→  // Run form
    90→  const [periodStart, setPeriodStart] = useState('')
    91→  const [periodEnd, setPeriodEnd] = useState('')
    92→  const [payDate, setPayDate] = useState('')
    93→
    94→  // Employee config form
    95→  const [cfgSalaryType, setCfgSalaryType] = useState('salary')
    96→  const [cfgSalaryAmount, setCfgSalaryAmount] = useState('')
    97→  const [cfgPayFrequency, setCfgPayFrequency] = useState('bi-weekly')
    98→  const [cfgPaymentMethod, setCfgPaymentMethod] = useState('direct_deposit')
    99→  const [cfgTaxStatus, setCfgTaxStatus] = useState('single')
   100→  const [cfgFederalAllow, setCfgFederalAllow] = useState('0')
   101→  const [cfgStateAllow, setCfgStateAllow] = useState('0')
   102→
   103→  useEffect(() => {
   104→    loadAll()
   105→  }, [])
   106→
   107→  async function loadAll() {
   108→    setLoading(true)
   109→    try {
   110→      const [dashRes, empRes] = await Promise.allSettled([
   111→        apiCall<DashboardData>('/payroll/dashboard'),
   112→        apiCall<{ employees: Employee[] }>('/payroll/employees'),
   113→      ])
   114→      if (dashRes.status === 'fulfilled') setDashboard(dashRes.value)
   115→      if (empRes.status === 'fulfilled') setEmployees(empRes.value.employees || [])
   116→    } catch {
   117→      // silent
   118→    } finally {
   119→      setLoading(false)
   120→    }
   121→  }
   122→
   123→  async function createPayrollRun() {
   124→    if (!periodStart || !periodEnd || !payDate) return
   125→    setCreating(true)
   126→    try {
   127→      await apiCall('/payroll/runs', {
   128→        method: 'POST',
   129→        body: { pay_period_start: periodStart, pay_period_end: periodEnd, pay_date: payDate },
   130→      })
   131→      setShowRunCreate(false)
   132→      setPeriodStart('')
   133→      setPeriodEnd('')
   134→      setPayDate('')
   135→      await loadAll()
   136→      setTab('runs')
   137→    } catch (err: unknown) {
   138→      alert(err instanceof Error ? err.message : 'Failed to create payroll run')
   139→    } finally {
   140→      setCreating(false)
   141→    }
   142→  }
   143→
   144→  async function viewRun(run: PayrollRun) {
   145→    setSelectedRun(run)
   146→    setLoadingRun(true)
   147→    try {
   148→      const res = await apiCall<{ payrollRun: PayrollRun; paychecks: Paycheck[] }>(`/payroll/runs/${run.id}`)
   149→      setRunPaychecks(res.paychecks || [])
   150→      setSelectedRun(res.payrollRun)
   151→    } catch {
   152→      setRunPaychecks([])
   153→    } finally {
   154→      setLoadingRun(false)
   155→    }
   156→  }
   157→
   158→  async function processRun(runId: number) {
   159→    if (!confirm('Process this payroll? This will mark all paychecks as paid.')) return
   160→    setProcessing(true)
   161→    try {
   162→      await apiCall(`/payroll/runs/${runId}/process`, { method: 'POST' })
   163→      setSelectedRun(null)
   164→      await loadAll()
   165→    } catch (err: unknown) {
   166→      alert(err instanceof Error ? err.message : 'Failed to process payroll')
   167→    } finally {
   168→      setProcessing(false)
   169→    }
   170→  }
   171→
   172→  function openEmployeeConfig(emp: Employee) {
   173→    setSelectedEmployee(emp)
   174→    setCfgSalaryType(emp.salary_type || 'salary')
   175→    setCfgSalaryAmount(String(emp.salary_amount || ''))
   176→    setCfgPayFrequency(emp.pay_frequency || 'bi-weekly')
   177→    setCfgPaymentMethod(emp.payment_method || 'direct_deposit')
   178→    setCfgTaxStatus('single')
   179→    setCfgFederalAllow('0')
   180→    setCfgStateAllow('0')
   181→  }
   182→
   183→  async function saveEmployeeConfig() {

   187→      await apiCall(`/payroll/employees/${selectedEmployee.id}/onboard`, {
   188→        method: 'POST',
   189→        body: {
   190→          salary_type: cfgSalaryType,
   191→          salary_amount: Number(cfgSalaryAmount),
   192→          pay_frequency: cfgPayFrequency,
   193→          payment_method: cfgPaymentMethod,
   194→          tax_filing_status: cfgTaxStatus,
   195→          federal_allowances: Number(cfgFederalAllow),
   196→          state_allowances: Number(cfgStateAllow),
   197→        },
   198→      })
   199→      setSelectedEmployee(null)
   200→      await loadAll()
   201→    } catch (err: unknown) {
   202→      alert(err instanceof Error ? err.message : 'Failed to save pay config')
   203→    } finally {
   204→      setSavingConfig(false)
   205→    }
   206→  }
   207→
   208→  const filteredEmployees = employees.filter(e =>
   209→    !search ||
   210→    e.employee_name?.toLowerCase().includes(search.toLowerCase()) ||
   211→    e.employee_email?.toLowerCase().includes(search.toLowerCase()) ||
   212→    e.employee_number?.toLowerCase().includes(search.toLowerCase())
   213→  )
   214→
   215→  const allRuns = [
   216→    ...(dashboard?.upcomingPayrolls || []),
   217→    ...(dashboard?.recentPayrolls || []),
   218→  ].filter((r, i, arr) => arr.findIndex(x => x.id === r.id) === i)
   219→    .sort((a, b) => new Date(b.pay_date).getTime() - new Date(a.pay_date).getTime())
   220→
   221→  if (loading) {
   222→    return (
   223→      <div className="flex items-center justify-center py-20">
   224→        <div className="h-8 w-8 animate-spin rounded-full border-2 border-primary border-t-transparent" />
   225→      </div>
   226→    )
   227→  }
   228→
   229→  return (
   230→    <div className="space-y-6">
   231→      <div className="flex items-center justify-between">
   232→        <div>
   233→          <h1 className="font-heading text-2xl font-bold">Payroll</h1>
   234→          <p className="text-muted-foreground">Manage employee compensation, run payroll, and generate pay stubs</p>
   235→        </div>
   236→        <Button onClick={() => setShowRunCreate(true)} className="gap-2">
   237→          <Play className="h-4 w-4" /> Run Payroll
   238→        </Button>
   239→      </div>
   240→
   241→      {/* Stats */}
   242→      <div className="grid gap-3 sm:grid-cols-4">
   243→        <Card>
   244→          <CardContent className="p-4">
   245→            <div className="flex items-center gap-3">
   246→              <div className="rounded-lg bg-blue-100 p-2">
   247→                <Users className="h-5 w-5 text-blue-600" />
   248→              </div>
   249→              <div>
   250→                <p className="text-2xl font-bold">{dashboard?.activeEmployees || 0}</p>
   251→                <p className="text-xs text-muted-foreground">Active Employees</p>
   252→              </div>
   253→            </div>
   254→          </CardContent>
   255→        </Card>
   256→        <Card>
   257→          <CardContent className="p-4">
   258→            <div className="flex items-center gap-3">
   259→              <div className="rounded-lg bg-emerald-100 p-2">
   260→                <DollarSign className="h-5 w-5 text-emerald-600" />
   261→              </div>
   262→              <div>
   263→                <p className="text-2xl font-bold">{fmt(dashboard?.monthlyTotal || 0)}</p>
   264→                <p className="text-xs text-muted-foreground">This Month</p>
   265→              </div>
   266→            </div>
   267→          </CardContent>
   268→        </Card>

   270→          <CardContent className="p-4">
   271→            <div className="flex items-center gap-3">
   272→              <div className="rounded-lg bg-amber-100 p-2">
   273→                <Clock className="h-5 w-5 text-amber-600" />
   274→              </div>
   275→              <div>
   276→                <p className="text-2xl font-bold">{dashboard?.upcomingPayrolls?.length || 0}</p>
   277→                <p className="text-xs text-muted-foreground">Upcoming Runs</p>
   278→              </div>
   279→            </div>
   280→          </CardContent>
   281→        </Card>
   282→        <Card>
   283→          <CardContent className="p-4">
   284→            <div className="flex items-center gap-3">
   285→              <div className="rounded-lg bg-purple-100 p-2">
   286→                <TrendingUp className="h-5 w-5 text-purple-600" />
   287→              </div>
   288→              <div>
   289→                <p className="text-2xl font-bold">{dashboard?.recentPayrolls?.length || 0}</p>
   290→                <p className="text-xs text-muted-foreground">Completed Runs</p>
   291→              </div>
   292→            </div>
   293→          </CardContent>
   294→        </Card>
   295→      </div>

   297→      {/* Tabs */}

   304→
   305→        {/* Overview Tab */}
   306→        <TabsContent value="dashboard">

   309→            <Card>
   310→              <CardContent className="p-5">

   313→                    <Calendar className="h-4 w-4 text-amber-500" /> Upcoming Payrolls
   314→                  </h3>
   315→                </div>
   316→                {(dashboard?.upcomingPayrolls || []).length === 0 ? (
   317→                  <div className="text-center py-8">

   321→                      <Play className="h-3 w-3" /> Schedule Run

   324→                ) : (
   325→                  <div className="space-y-3">
   326→                    {dashboard!.upcomingPayrolls.map(run => (
   327→                      <div

   335→                          </p>
   336→                          <p className="text-xs text-muted-foreground">
   337→                            Pay date: {new Date(run.pay_date).toLocaleDateString()} · {run.employee_count || 0} employees
   338→                          </p>
   339→                        </div>
   340→                        <Badge variant={run.status === 'draft' ? 'warning' : 'default'}>
   341→                          {run.status}

   343→                      </div>

   347→              </CardContent>
   348→            </Card>
   349→
   350→            {/* Recent payrolls */}
   351→            <Card>
   352→              <CardContent className="p-5">
   353→                <div className="flex items-center justify-between mb-4">

   360→                    <Receipt className="mx-auto mb-2 h-8 w-8 opacity-20" />

   369→                        onClick={() => viewRun(run)}
   370→                      >
   371→                        <div>
   372→                          <p className="font-medium text-sm">
   373→                            {new Date(run.pay_period_start).toLocaleDateString()} – {new Date(run.pay_period_end).toLocaleDateString()}

   375→                          <p className="text-xs text-muted-foreground">
   376→                            Paid {new Date(run.pay_date).toLocaleDateString()} · {fmt(run.total_net)} net
