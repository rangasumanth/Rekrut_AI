     6→// 2026 Federal Tax Brackets (Single)
     7→const FEDERAL_TAX_BRACKETS = [
     8→  { min: 0, max: 11600, rate: 0.10 },
     9→  { min: 11600, max: 47150, rate: 0.12 },
    10→  { min: 47150, max: 100525, rate: 0.22 },
    11→  { min: 100525, max: 191950, rate: 0.24 },

    17→// Social Security and Medicare rates
    18→const SOCIAL_SECURITY_RATE = 0.062;
    19→const SOCIAL_SECURITY_WAGE_BASE = 168600; // 2026 limit
    20→const MEDICARE_RATE = 0.0145;
    21→const ADDITIONAL_MEDICARE_RATE = 0.009; // Additional 0.9% over $200k

    23→/**
    24→ * Calculate gross pay based on salary configuration
    25→ */
    26→function calculateGrossPay(config, hoursWorked = null) {
    27→  const { salary_type, salary_amount, pay_frequency } = config;
    28→
    29→  if (salary_type === 'hourly' && hoursWorked) {
    30→    return hoursWorked * parseFloat(salary_amount);
    31→  }
    32→
    33→  // Salary - divide by pay periods per year
    34→  const periodsPerYear = {
    35→    'weekly': 52,
    36→    'bi-weekly': 26,
    37→    'semi-monthly': 24,
    38→    'monthly': 12
    39→  };
    40→
    41→  const periods = periodsPerYear[pay_frequency] || 26;
    42→  return parseFloat(salary_amount) / periods;
    43→}

    45→/**
    46→ * Calculate federal income tax withholding
    47→ */
    48→function calculateFederalTax(grossPay, filingStatus = 'single', allowances = 0) {
    49→  // Annualize the gross pay (assuming bi-weekly)
    50→  const annualizedGross = grossPay * 26;
    51→
    52→  // Standard deduction for 2026 (simplified)
    53→  const standardDeduction = filingStatus === 'single' ? 14600 : 29200;
    54→  const allowanceAmount = allowances * 4700; // Per allowance
    55→
    56→  let taxableIncome = Math.max(0, annualizedGross - standardDeduction - allowanceAmount);
    57→
    58→  // Calculate tax using brackets
    59→  let tax = 0;
    60→  for (let bracket of FEDERAL_TAX_BRACKETS) {
    61→    if (taxableIncome <= bracket.min) break;
    62→
    63→    const taxableInBracket = Math.min(
    64→      taxableIncome - bracket.min,
    65→      bracket.max - bracket.min
    66→    );
    67→
    68→    tax += taxableInBracket * bracket.rate;
    69→  }
    70→
    71→  // Convert annual tax back to pay period
    72→  return tax / 26;
    73→}

    76→ * Calculate state income tax (simplified - using CA as example)

    78→function calculateStateTax(grossPay, state = 'CA') {
    79→  // Simplified state tax calculation (using CA as example: ~5% average)
    80→  // In production, this would use actual state tax tables
    81→  const stateRates = {
    82→    'CA': 0.05,
    83→    'NY': 0.045,
    84→    'TX': 0.00,
    85→    'FL': 0.00,
    86→    'WA': 0.00
    87→  };
    88→
    89→  const rate = stateRates[state] || 0.04;
    90→  return grossPay * rate;
    91→}

    93→/**
    94→ * Calculate Social Security tax
    95→ */
    96→function calculateSocialSecurity(grossPay, ytdGross = 0) {
    97→  const remaining = Math.max(0, SOCIAL_SECURITY_WAGE_BASE - ytdGross);
    98→  const taxableAmount = Math.min(grossPay, remaining);
    99→  return taxableAmount * SOCIAL_SECURITY_RATE;
   100→}

   102→/**
   103→ * Calculate Medicare tax
   104→ */
   105→function calculateMedicare(grossPay, ytdGross = 0) {
   106→  let medicare = grossPay * MEDICARE_RATE;
   107→
   108→  // Additional Medicare tax on income over $200k
   109→  const thresholdRemaining = Math.max(0, 200000 - ytdGross);
   110→  if (grossPay > thresholdRemaining) {
   111→    const additionalTaxableAmount = grossPay - thresholdRemaining;
   112→    medicare += additionalTaxableAmount * ADDITIONAL_MEDICARE_RATE;
   113→  }
   114→
   115→  return medicare;
   116→}

   118→/**
   119→ * Calculate complete paycheck
   120→ */
   121→function calculatePaycheck(employee, config, hoursWorked = null, ytdGross = 0) {
   122→  const grossPay = calculateGrossPay(config, hoursWorked);
   123→
   124→  const federalTax = calculateFederalTax(
   125→    grossPay,
   126→    config.tax_filing_status,
   127→    config.federal_allowances
   128→  );
   129→
   130→  const stateTax = calculateStateTax(grossPay, 'CA'); // Default to CA for now
   131→
   132→  const socialSecurity = calculateSocialSecurity(grossPay, ytdGross);
   133→  const medicare = calculateMedicare(grossPay, ytdGross);
   134→
   135→  const additionalWithholding = parseFloat(config.additional_withholding || 0);
   136→
   137→  const totalDeductions =
   138→    federalTax +
   139→    stateTax +
   140→    socialSecurity +
   141→    medicare +
   142→    additionalWithholding;
   143→
   144→  const netPay = grossPay - totalDeductions;
   145→
   146→  return {
   147→    grossPay: parseFloat(grossPay.toFixed(2)),
   148→    federalTax: parseFloat(federalTax.toFixed(2)),
   149→    stateTax: parseFloat(stateTax.toFixed(2)),
   150→    socialSecurity: parseFloat(socialSecurity.toFixed(2)),
   151→    medicare: parseFloat(medicare.toFixed(2)),
   152→    otherDeductions: parseFloat(additionalWithholding.toFixed(2)),
   153→    totalDeductions: parseFloat(totalDeductions.toFixed(2)),
   154→    netPay: parseFloat(netPay.toFixed(2)),
   155→    hoursWorked: hoursWorked
   156→  };
   157→}

   159→/**
   160→ * Calculate paycheck for non-US countries
   161→ * Uses simplified models per country (production would integrate actual tax APIs)
   162→ */
   163→function calculateInternationalPaycheck(employee, config, countryCode, hoursWorked = null) {
   164→  const grossPay = calculateGrossPay(config, hoursWorked);
   165→
   166→  const calculators = {
   167→    // India: Income tax slabs + PF + ESI
   168→    IN: () => {
   169→      const annualized = grossPay * 12;
   170→      let incomeTax = 0;
   171→      // New Tax Regime (2025-26)
   172→      const slabs = [
   173→        { min: 0, max: 300000, rate: 0 },
   174→        { min: 300000, max: 700000, rate: 0.05 },
   175→        { min: 700000, max: 1000000, rate: 0.10 },
   176→        { min: 1000000, max: 1200000, rate: 0.15 },
   177→        { min: 1200000, max: 1500000, rate: 0.20 },
   178→        { min: 1500000, max: Infinity, rate: 0.30 },
   179→      ];
   180→      for (const slab of slabs) {
   181→        if (annualized <= slab.min) break;
   182→        incomeTax += Math.min(annualized - slab.min, slab.max - slab.min) * slab.rate;
   183→      }
   184→      const monthlyTax = incomeTax / 12;
   185→      const pfEmployee = grossPay * 0.12; // 12% employee PF
   186→      const esiEmployee = grossPay <= 21000 ? grossPay * 0.0075 : 0; // 0.75% ESI
   187→      const professionalTax = 200; // Max ₹200/month in most states
   188→      const totalDeductions = monthlyTax + pfEmployee + esiEmployee + professionalTax;
   189→      return {
   190→        grossPay: parseFloat(grossPay.toFixed(2)),
   191→        incomeTax: parseFloat(monthlyTax.toFixed(2)),
   192→        providentFund: parseFloat(pfEmployee.toFixed(2)),
   193→        esi: parseFloat(esiEmployee.toFixed(2)),
   194→        professionalTax,
   195→        totalDeductions: parseFloat(totalDeductions.toFixed(2)),
   196→        netPay: parseFloat((grossPay - totalDeductions).toFixed(2)),
   197→        country: 'IN', currency: 'INR',
   198→      };
   199→    },
   200→    // UK: PAYE + National Insurance
   201→    GB: () => {
   202→      const annualized = grossPay * 12;
   203→      const personalAllowance = 12570;
   204→      const taxable = Math.max(0, annualized - personalAllowance);
   205→      let incomeTax = 0;
   206→      if (taxable <= 37700) incomeTax = taxable * 0.20;
   207→      else if (taxable <= 125140) incomeTax = 37700 * 0.20 + (taxable - 37700) * 0.40;
   208→      else incomeTax = 37700 * 0.20 + (125140 - 37700) * 0.40 + (taxable - 125140) * 0.45;
   209→      const monthlyTax = incomeTax / 12;
   210→      // NI Class 1: 8% on earnings between £12,570-£50,270, 2% above
   211→      const niThreshold = 12570 / 12;
   212→      const niUpper = 50270 / 12;
   213→      let ni = 0;
   214→      if (grossPay > niThreshold) ni = Math.min(grossPay - niThreshold, niUpper - niThreshold) * 0.08;
   215→      if (grossPay > niUpper) ni += (grossPay - niUpper) * 0.02;
   216→      const totalDeductions = monthlyTax + ni;
   217→      return {
   218→        grossPay: parseFloat(grossPay.toFixed(2)),
   219→        incomeTax: parseFloat(monthlyTax.toFixed(2)),
   220→        nationalInsurance: parseFloat(ni.toFixed(2)),
   221→        totalDeductions: parseFloat(totalDeductions.toFixed(2)),
   222→        netPay: parseFloat((grossPay - totalDeductions).toFixed(2)),
   223→        country: 'GB', currency: 'GBP',
   224→      };
   225→    },
   226→    // Canada: Federal + Provincial tax + CPP + EI
   227→    CA: () => {
   228→      const annualized = grossPay * (config.pay_frequency === 'monthly' ? 12 : 26);
   229→      const basicPersonal = 15705;
   230→      const taxable = Math.max(0, annualized - basicPersonal);
   231→      let federalTax = 0;
   232→      const brackets = [
   233→        { min: 0, max: 55867, rate: 0.15 },
   234→        { min: 55867, max: 111733, rate: 0.205 },
   235→        { min: 111733, max: 154906, rate: 0.26 },
   236→        { min: 154906, max: 220000, rate: 0.29 },
   237→        { min: 220000, max: Infinity, rate: 0.33 },
   238→      ];
   239→      for (const b of brackets) {
   240→        if (taxable <= b.min) break;
   241→        federalTax += Math.min(taxable - b.min, b.max - b.min) * b.rate;
   242→      }
   243→      const periods = config.pay_frequency === 'monthly' ? 12 : 26;
   244→      const periodTax = federalTax / periods;
   245→      const provincialTax = periodTax * 0.5; // Simplified: ~50% of federal
   246→      const cpp = Math.min(grossPay * 0.0595, 3867.50 / periods);
   247→      const ei = Math.min(grossPay * 0.0166, 1049.12 / periods);
   248→      const totalDeductions = periodTax + provincialTax + cpp + ei;
   249→      return {
   250→        grossPay: parseFloat(grossPay.toFixed(2)),
   251→        federalTax: parseFloat(periodTax.toFixed(2)),
   252→        provincialTax: parseFloat(provincialTax.toFixed(2)),
   253→        cpp: parseFloat(cpp.toFixed(2)),
   254→        ei: parseFloat(ei.toFixed(2)),
   255→        totalDeductions: parseFloat(totalDeductions.toFixed(2)),
   256→        netPay: parseFloat((grossPay - totalDeductions).toFixed(2)),
   257→        country: 'CA', currency: 'CAD',
   258→      };
   259→    },
   260→  };
   261→
   262→  if (calculators[countryCode]) return calculators[countryCode]();
   263→
   264→  // Fallback: simple flat 25% estimated tax
   265→  const estimatedTax = grossPay * 0.25;
   266→  return {
   267→    grossPay: parseFloat(grossPay.toFixed(2)),
   268→    estimatedTax: parseFloat(estimatedTax.toFixed(2)),
   269→    totalDeductions: parseFloat(estimatedTax.toFixed(2)),
   270→    netPay: parseFloat((grossPay - estimatedTax).toFixed(2)),
