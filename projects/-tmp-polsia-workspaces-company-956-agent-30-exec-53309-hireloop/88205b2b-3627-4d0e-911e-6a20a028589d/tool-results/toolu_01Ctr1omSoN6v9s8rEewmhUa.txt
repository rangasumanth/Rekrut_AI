     9→// Handle OAuth callback tokens from URL
    10→function handleOAuthCallback() {
    11→  const params = new URLSearchParams(window.location.search);
    12→  const token = params.get('token');
    13→  const refresh = params.get('refresh');
    14→
    15→  if (token) {
    16→    localStorage.setItem(TOKEN_KEY, token);
    17→    if (refresh) {
    18→      localStorage.setItem(REFRESH_KEY, refresh);
    19→    }
    20→    // Clean URL
    21→    window.history.replaceState({}, document.title, window.location.pathname);
    22→  }
    23→}

    25→// Refresh access token using refresh token
    26→async function refreshAccessToken() {
    27→  const refreshToken = localStorage.getItem(REFRESH_KEY);
    28→  if (!refreshToken) return null;
    29→
    30→  try {
    31→    const response = await fetch('/api/auth/refresh', {
    32→      method: 'POST',
    33→      headers: { 'Content-Type': 'application/json' },
    34→      body: JSON.stringify({ refreshToken })
    35→    });
    36→
    37→    if (response.ok) {
    38→      const data = await response.json();
    39→      if (data.success) {
    40→        localStorage.setItem(TOKEN_KEY, data.accessToken);
    41→        localStorage.setItem(REFRESH_KEY, data.refreshToken);
    42→        return data.accessToken;
    43→      }
    44→    }
    45→  } catch (err) {
    46→    console.error('Token refresh failed:', err);
    47→  }
    48→
    49→  // Refresh failed - clear tokens
    50→  localStorage.removeItem(TOKEN_KEY);
    51→  localStorage.removeItem(REFRESH_KEY);
    52→  return null;
    53→}

    55→// Check auth status on page load
    56→// Uses a promise guard to prevent race conditions when called from multiple DOMContentLoaded handlers
    57→async function checkAuth() {
    58→  // If already running, return the existing promise (prevents double token refresh)
    59→  if (_checkAuthPromise) return _checkAuthPromise;
    60→
    61→  _checkAuthPromise = _doCheckAuth();
    62→  try {
    63→    return await _checkAuthPromise;
    64→  } finally {
    65→    _checkAuthPromise = null;
    66→  }
    67→}
    68→
    69→async function _doCheckAuth() {
    70→  // Handle OAuth callback first
    71→  handleOAuthCallback();
    72→
    73→  const token = localStorage.getItem(TOKEN_KEY);
    74→  if (!token) return null;
    75→
    76→  try {
    77→    const response = await fetch('/api/auth/me', {
    78→      headers: { 'Authorization': `Bearer ${token}` }
    79→    });
    80→
    81→    if (response.ok) {
    82→      const data = await response.json();
    83→      currentUser = data.user;
    84→      updateNavForAuth();
    85→      return data.user;
    86→    }
    87→
    88→    // Token expired - try refresh
    89→    if (response.status === 401) {
    90→      const newToken = await refreshAccessToken();
    91→      if (newToken) {
    92→        // Retry with new token (reset promise guard for retry)
    93→        _checkAuthPromise = null;
    94→        return await checkAuth();
    95→      }
    96→    }
    97→  } catch (err) {
    98→    console.error('Auth check failed:', err);
    99→  }
   100→  return null;
   101→}

   120→// Registration
   121→async function handleRegister(event) {
   122→  event.preventDefault();
   123→  const form = event.target;
   124→  const submitBtn = form.querySelector('button[type="submit"]');
   125→  const errorEl = document.getElementById('register-error');
   126→
   127→  submitBtn.disabled = true;
   128→  submitBtn.textContent = 'Creating account...';
   129→  if (errorEl) errorEl.style.display = 'none';
   130→
   131→  try {
   132→    const response = await fetch('/api/auth/register', {
   133→      method: 'POST',
   134→      headers: { 'Content-Type': 'application/json' },
   135→      body: JSON.stringify({
   136→        email: form.email.value,
   137→        password: form.password.value,
   138→        name: form.name?.value || '',
   139→        role: form.role?.value || 'candidate',
   140→        company_name: form.company_name?.value || ''
   141→      })
   142→    });
   143→
   144→    const data = await response.json();
   145→
   146→    if (data.success) {
   147→      localStorage.setItem(TOKEN_KEY, data.accessToken || data.token);
   148→      if (data.refreshToken) {
   149→        localStorage.setItem(REFRESH_KEY, data.refreshToken);
   150→      }
   151→      // Redirect based on role
   152→      const role = form.role?.value || 'candidate';
   153→      window.location.href = role === 'recruiter'
   154→        ? '/recruiter-dashboard.html'
   155→        : '/candidate-dashboard.html';
   156→    } else {
   157→      if (errorEl) {
   158→        errorEl.textContent = data.error || 'Registration failed';
   159→        errorEl.style.display = 'block';
   160→      }
   161→    }
   162→  } catch (err) {
   163→    if (errorEl) {
   164→      errorEl.textContent = 'Network error. Please try again.';
   165→      errorEl.style.display = 'block';
   166→    }
   167→  } finally {
   168→    submitBtn.disabled = false;
   169→    submitBtn.textContent = 'Create Account';
   170→  }
   171→}

   240→// API helper with automatic token refresh
   241→async function apiCall(endpoint, options = {}) {
   242→  let token = localStorage.getItem(TOKEN_KEY);
   243→  const headers = {
   244→    'Content-Type': 'application/json',
   245→    ...(token && { 'Authorization': `Bearer ${token}` }),
   246→    ...options.headers
   247→  };
   248→
   249→  let response = await fetch(`/api${endpoint}`, {
   250→    ...options,
   251→    headers
   252→  });
   253→
   254→  // If 401 and we have a refresh token, try to refresh
   255→  if (response.status === 401) {
   256→    const newToken = await refreshAccessToken();
   257→    if (newToken) {
   258→      // Retry with new token
   259→      headers['Authorization'] = `Bearer ${newToken}`;
   260→      response = await fetch(`/api${endpoint}`, {
   261→        ...options,
   262→        headers
   263→      });
   264→    } else {
   265→      // Still unauthorized - redirect to login
   266→      window.location.href = '/login.html';
   267→      return null;
   268→    }
   269→  }
   270→
   271→  return response.json();
   272→}

   352→// Mobile Menu Toggle
   353→function initMobileMenu() {
   354→  const navLinks = document.getElementById('nav-links');
   355→  const navContainer = document.querySelector('.nav-container');
   356→
   357→  // Create hamburger button if it doesn't exist
   358→  let mobileToggle = document.querySelector('.mobile-menu-toggle');
   359→  if (!mobileToggle && navLinks && navContainer) {
   360→    mobileToggle = document.createElement('button');
   361→    mobileToggle.className = 'mobile-menu-toggle';
   362→    mobileToggle.innerHTML = '☰';
   363→    mobileToggle.setAttribute('aria-label', 'Toggle menu');
   364→
   365→    // Insert before nav-auth
   366→    const navAuth = document.getElementById('nav-auth');
   367→    if (navAuth) {
   368→      navContainer.insertBefore(mobileToggle, navAuth);
   369→    }
   370→  }
   371→
   372→  if (mobileToggle && navLinks) {
   373→    // Toggle mobile menu
   374→    mobileToggle.addEventListener('click', (e) => {
   375→      e.stopPropagation();
   376→      navLinks.classList.toggle('mobile-open');
   377→      mobileToggle.innerHTML = navLinks.classList.contains('mobile-open') ? '✕' : '☰';
   378→    });
   379→
   380→    // Close menu when clicking outside
   381→    document.addEventListener('click', (e) => {
   382→      if (!navLinks.contains(e.target) && !mobileToggle.contains(e.target)) {
   383→        navLinks.classList.remove('mobile-open');
   384→        mobileToggle.innerHTML = '☰';
   385→      }
   386→    });
   387→
   388→    // Handle dropdown toggles on mobile
   389→    const dropdowns = navLinks.querySelectorAll('.nav-dropdown');
   390→    dropdowns.forEach(dropdown => {
   391→      const toggle = dropdown.querySelector('.nav-dropdown-toggle');
   392→      if (toggle) {
   393→        toggle.addEventListener('click', (e) => {
   394→          if (window.innerWidth <= 768) {
   395→            e.preventDefault();
   396→            dropdown.classList.toggle('active');
   397→          }
   398→        });
   399→      }
   400→    });
   401→  }
   402→}
