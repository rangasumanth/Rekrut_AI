   117→// Score color helper
   118→function scoreColor(score: number): string {
   119→  if (score >= 8) return 'text-green-600'
   120→  if (score >= 6) return 'text-amber-600'
   121→  return 'text-red-600'
   122→}
   123→
   124→function scoreBg(score: number): string {
   125→  if (score >= 8) return 'bg-green-50 border-green-200'
   126→  if (score >= 6) return 'bg-amber-50 border-amber-200'
   127→  return 'bg-red-50 border-red-200'
   128→}

   130→function scoreLabel(score: number): string {
   131→  if (score >= 9) return 'Excellent'
   132→  if (score >= 8) return 'Great'
   133→  if (score >= 7) return 'Good'
   134→  if (score >= 6) return 'Decent'
   135→  if (score >= 5) return 'Average'
   136→  return 'Needs Work'
   137→}

   140→function ScoreBar({ score, label, icon: Icon }: { score: number; label: string; icon: React.ElementType }) {
   141→  const pct = (score / 10) * 100
   142→  const barColor = score >= 8 ? 'bg-green-500' : score >= 6 ? 'bg-amber-500' : 'bg-red-500'
   143→

   145→    <div className="space-y-1">

   147→        <span className="flex items-center gap-1.5 font-medium">
   148→          <Icon className="h-3.5 w-3.5 text-muted-foreground" />
   149→          {label}
   150→        </span>
   151→        <span className={`font-bold ${scoreColor(score)}`}>{score}/10</span>

   342→  //
   343→  // FIX: Trust the MediaStreamTrack's getSettings() instead of polling
   344→  // video.readyState. If the track is 'live' with valid width/height,
   345→  // the camera IS working. Return true immediately so the caller can
   346→  // set cameraReady=true, which removes the overlay, which lets iOS
   347→  // start rendering.
   348→  async function attachStreamToVideo(stream: MediaStream): Promise<boolean> {
   349→    const video = videoRef.current
   350→    if (!video) {
   351→      console.error('[camera] video ref not available')
   352→      setCameraStatus('No video element')
   353→      return false
   354→    }
   355→
   356→    // Detach first (clean slate)
   357→    video.srcObject = null
   358→    await new Promise(r => setTimeout(r, 50))
   359→    video.srcObject = stream
   360→
   361→    // Explicit play() — on iOS autoPlay can be unreliable after getUserMedia
   362→    try {
   363→      await video.play()
   364→    } catch (playErr: any) {
   365→      console.warn('[camera] play() error:', playErr?.message)
   366→      // Don't fail — muted+playsInline videos may auto-start
   367→    }
   368→
   369→    // ── PRIMARY CHECK: Trust the track's settings ──
   370→    // If the track is live and reports valid dimensions, the camera IS active.
   371→    // Don't wait for video.readyState (which won't update while overlay covers video on iOS).
   372→    const vt = stream.getVideoTracks()[0]
   373→    if (vt && vt.readyState === 'live') {
   374→      const settings = vt.getSettings?.() || {}
   375→      if ((settings.width || 0) > 0 && (settings.height || 0) > 0) {
   376→        console.log(`[camera] track live with valid settings ${settings.width}x${settings.height} — trusting track (iOS fix)`)
   377→        setCameraStatus(`Live ${settings.width}x${settings.height}`)
   378→        return true
   379→      }
   380→    }
   381→
   382→    // ── FALLBACK: Poll for frame data OR track settings (up to 4 seconds) ──
   383→    // Handles edge cases where track settings aren't immediately available
   384→    return new Promise<boolean>((resolve) => {
   385→      const start = Date.now()
   386→      const check = () => {
   387→        const v = videoRef.current
   388→        if (!v || v.srcObject !== stream) { resolve(false); return }
   389→
   390→        const elapsed = Date.now() - start
   391→
   392→        // Check video element (works on desktop / non-overlay scenarios)
   393→        if (v.readyState >= 2 && v.videoWidth > 0) {
   394→          console.log(`[camera] frames confirmed after ${elapsed}ms — ${v.videoWidth}x${v.videoHeight}`)
   395→          resolve(true)
   396→          return
   397→        }
   398→
   399→        // Re-check track settings (may become available after a delay)
   400→        const vt2 = stream.getVideoTracks()[0]
   401→        if (vt2 && vt2.readyState === 'live' && elapsed > 300) {
   402→          const s = vt2.getSettings?.() || {}
   403→          if ((s.width || 0) > 0 && (s.height || 0) > 0) {
   404→            console.log(`[camera] track settings available after ${elapsed}ms: ${s.width}x${s.height} — trusting`)
   405→            setCameraStatus(`Live ${s.width}x${s.height}`)
   406→            resolve(true)
   407→            return
   408→          }
   409→        }
   410→
   411→        // Also accept if track is live and enabled (even without settings)
   412→        if (vt2 && vt2.readyState === 'live' && vt2.enabled && !vt2.muted && elapsed > 1500) {
   413→          console.log(`[camera] track live+enabled after ${elapsed}ms — accepting without settings`)
   414→          setCameraStatus('Live (no settings)')
   415→          resolve(true)
   416→          return
   417→        }
   418→
   419→        // Diagnostic info
   420→        const diag = vt2
   421→          ? `trk:${vt2.readyState} muted:${vt2.muted} enabled:${vt2.enabled}`
   422→          : 'no-vt'
   423→        setCameraStatus(`Waiting ${Math.round(elapsed / 1000)}s rs=${v.readyState} ${diag}`)
   424→
   425→        if (elapsed > 4000) {
   426→          console.warn('[camera] no frames or track settings after 4s')
   427→          resolve(false)
   428→          return
   429→        }
   430→        setTimeout(check, 200)
   431→      }
   432→      setTimeout(check, 100)
   433→    })
   434→  }

   436→  async function startCamera() {
   437→    try {
   438→      setCameraError(null)
   439→      setCameraReady(false)
   440→      setCameraStatus('Requesting camera...')
   441→
   442→      if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
   443→        setCameraError('not_supported')
   444→        setCameraStatus('Camera not supported')
   445→        return
   446→      }
   447→
   448→      // ── STEP 1: Request VIDEO-ONLY (forces camera permission prompt on iOS) ──
   449→      let videoStream: MediaStream | null = null
   450→      const constraintSets = [
   451→        { video: { facingMode: 'user' }, label: 'video:user' },
   452→        { video: true, label: 'video:true' },
   453→      ]
   454→
   455→      for (const { video: vc, label } of constraintSets) {
   456→        try {
   457→          setCameraStatus(`Trying ${label}...`)
   458→          // @ts-ignore — vc can be boolean or object
   459→          videoStream = await navigator.mediaDevices.getUserMedia({ video: vc })
   460→
   461→          const vt = videoStream.getVideoTracks()[0]
   462→          if (!vt) {
   463→            console.warn(`[camera] ${label}: no video track returned`)
   464→            videoStream.getTracks().forEach(t => t.stop())
   465→            videoStream = null
   466→            continue
   467→          }
   468→
   469→          // Check track has real settings (width/height > 0 means camera is actually active)
   470→          const settings = vt.getSettings?.() || {}
   471→          console.log(`[camera] ${label}: track=${vt.readyState} muted=${vt.muted} ${settings.width}x${settings.height}`)
   472→          setCameraStatus(`Got ${label}: ${settings.width || '?'}x${settings.height || '?'}`)
   473→
   474→          // Attach and verify frames flow
   475→          const framesOk = await attachStreamToVideo(videoStream)
   476→          if (framesOk) {
   477→            setCameraStatus(`Video OK via ${label}`)
   478→            break // Success!
   479→          }
   480→
   481→          // Frames didn't flow — try reattach once (srcObject can be finicky on iOS)
   482→          console.warn(`[camera] ${label}: no frames, retrying attach...`)
   483→          setCameraStatus(`Retry attach ${label}...`)
   484→          const retryOk = await attachStreamToVideo(videoStream)
   485→          if (retryOk) {
   486→            setCameraStatus(`Video OK via ${label} (retry)`)
   487→            break
   488→          }
   489→
   490→          // This constraint set failed
   491→          console.warn(`[camera] ${label}: failed after retry`)
   492→          videoStream.getTracks().forEach(t => t.stop())
   493→          videoStream = null
   494→        } catch (err: any) {
   495→          console.warn(`[camera] ${label} error: ${err?.name} ${err?.message}`)
   496→          setCameraStatus(`${label}: ${err?.name}`)
   497→          if (err.name === 'NotAllowedError') {
   498→            setCameraError('denied')
   499→            return
   500→          }
   501→          // Try next constraint set for OverconstrainedError, NotFoundError, etc.
   502→        }
   503→      }
   504→
   505→      if (!videoStream) {
   506→        // Last resort: try combined video+audio (some devices need it)
   507→        try {
   508→          setCameraStatus('Trying video+audio...')
   509→          videoStream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true })
   510→          const framesOk = await attachStreamToVideo(videoStream)
   511→          if (!framesOk) {
   512→            videoStream.getTracks().forEach(t => t.stop())
   513→            videoStream = null
   514→          } else {
   515→            setCameraStatus('Video OK via combined')
   516→          }
   517→        } catch (err: any) {
   518→          console.warn('[camera] combined fallback error:', err?.name)
   519→          if (err.name === 'NotAllowedError') {
   520→            setCameraError('denied')
   521→          } else {
   522→            setCameraError('not_found')
   523→          }
   524→          setCameraStatus(`All methods failed: ${err?.name}`)
   525→          return
   526→        }
   527→      }
   528→
   529→      if (!videoStream) {
   530→        setCameraError('not_found')
   531→        setCameraStatus('Camera not working — tap Retry')
   532→        return
   533→      }
   534→
   535→      // ── STEP 2: Add audio track (separate request) ──
   536→      const hasAudio = videoStream.getAudioTracks().length > 0
   537→      if (!hasAudio) {
   538→        try {
   539→          setCameraStatus('Adding microphone...')
   540→          const audioStream = await navigator.mediaDevices.getUserMedia({ audio: true })
   541→          const audioTrack = audioStream.getAudioTracks()[0]
   542→          if (audioTrack) {
   543→            videoStream.addTrack(audioTrack)
   544→            console.log('[camera] audio track added')
   545→          }
   546→        } catch (audioErr: any) {
   547→          // Audio failure is non-fatal — recording can work without mic
   548→          console.warn('[camera] audio request failed:', audioErr?.name)
   549→        }
   550→      }
   551→
   552→      // ── STEP 3: Finalize ──
   553→      streamRef.current = videoStream
   554→      const video = videoRef.current
   555→      if (video) {
   556→        // Ensure video element has the final stream (might have been set in attachStreamToVideo)
   557→        if (video.srcObject !== videoStream) {

   634→  // Speech Recognition
   635→  function startSpeechRecognition() {
   636→    const SpeechRecognition = (window as any).SpeechRecognition || (window as any).webkitSpeechRecognition
   637→    if (!SpeechRecognition) {
   638→      console.warn('Speech recognition not supported')
   639→      return
   640→    }
   641→
   642→    const recognition = new SpeechRecognition()
   643→    recognition.continuous = true
   644→    recognition.interimResults = true
   645→    recognition.lang = 'en-US'
   646→
   647→    let finalTranscript = ''
   648→
   649→    recognition.onresult = (event: any) => {
   650→      let interim = ''
   651→      for (let i = event.resultIndex; i < event.results.length; i++) {
   652→        const result = event.results[i]
   653→        if (result.isFinal) {
   654→          finalTranscript += result[0].transcript + ' '
   655→        } else {
   656→          interim = result[0].transcript
   657→        }
   658→      }
   659→      setTranscription(finalTranscript + interim)
   660→      setIsTranscribing(true)
   661→    }
   662→
   663→    recognition.onerror = (event: any) => {
   664→      console.warn('Speech recognition error:', event.error)
   665→      if (event.error === 'no-speech') {
   666→        // Restart
   667→        try { recognition.start() } catch (_) {}
   668→      }
   669→    }
   670→
   671→    recognition.onend = () => {
   672→      setIsTranscribing(false)
   673→      // Restart if still recording
   674→      if (isRecording) {
   675→        try { recognition.start() } catch (_) {}
   676→      }
   677→    }
   678→
   679→    recognition.start()
   680→    recognitionRef.current = recognition
   681→    setIsTranscribing(true)
   682→  }
