   229→/**
   230→ * Calculate cosine similarity between two vectors
   231→ */
   232→function cosineSimilarity(vecA, vecB) {
   233→  if (!vecA || !vecB || vecA.length !== vecB.length) {
   234→    return 0;
   235→  }
   236→
   237→  let dotProduct = 0;
   238→  let normA = 0;
   239→  let normB = 0;
   240→
   241→  for (let i = 0; i < vecA.length; i++) {
   242→    dotProduct += vecA[i] * vecB[i];
   243→    normA += vecA[i] * vecA[i];
   244→    normB += vecB[i] * vecB[i];
   245→  }
   246→
   247→  const magnitude = Math.sqrt(normA) * Math.sqrt(normB);
   248→  return magnitude === 0 ? 0 : dotProduct / magnitude;
   249→}
