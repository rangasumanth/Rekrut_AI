    60→  useEffect(() => {

    64→  async function loadApplications() {
    65→    try {
    66→      const data = await apiCall<{ success: boolean; applications: Application[] }>('/candidate/applications')

    68→    } catch {
    69→      // silent
    70→    } finally {
    71→      setLoading(false)
    72→    }

    75→  async function withdrawApplication() {
    76→    if (!withdrawTarget) return

    78→    try {
    79→      await apiCall(`/candidate/applications/${withdrawTarget.id}/withdraw`, {
    80→        method: 'PUT',
    81→      })
    82→      setApplications(prev =>
    83→        prev.map(a => a.id === withdrawTarget.id ? { ...a, status: 'withdrawn' } : a)
    84→      )
    85→      setWithdrawTarget(null)
    86→      setSelectedApp(null)
    87→    } catch (err: unknown) {
    88→      alert(err instanceof Error ? err.message : 'Failed to withdraw application')
    89→    } finally {
    90→      setWithdrawing(false)
    91→    }

    94→  function timeAgo(dateStr: string) {

    97→    if (days === 0) return 'Today'
    98→    if (days === 1) return 'Yesterday'
    99→    if (days < 7) return `${days} days ago`
   100→    if (days < 30) return `${Math.floor(days / 7)} weeks ago`
   101→    return `${Math.floor(days / 30)} months ago`
   102→  }

   136→      {/* Stats */}

   139→          <CardContent className="p-4 text-center">
   140→            <p className="text-2xl font-bold">{applications.length}</p>

   143→        </Card>
   144→        <Card>
   145→          <CardContent className="p-4 text-center">
   146→            <p className="text-2xl font-bold text-blue-600">
   147→              {applications.filter(a => !['rejected', 'withdrawn', 'hired'].includes(a.status)).length}
   148→            </p>
   149→            <p className="text-xs text-muted-foreground">Active</p>
   150→          </CardContent>
   151→        </Card>
   152→        <Card>
   153→          <CardContent className="p-4 text-center">
   154→            <p className="text-2xl font-bold text-amber-600">
   155→              {applications.filter(a => ['reviewing', 'interviewed'].includes(a.status)).length}
   156→            </p>
   157→            <p className="text-xs text-muted-foreground">In Progress</p>
   158→          </CardContent>
   159→        </Card>
   160→        <Card>
   161→          <CardContent className="p-4 text-center">
   162→            <p className="text-2xl font-bold text-emerald-600">
   163→              {applications.filter(a => ['offered', 'hired'].includes(a.status)).length}

   166→          </CardContent>
   167→        </Card>
   168→      </div>
   169→
   170→      {/* Status filter pills */}

   175→          onClick={() => setStatusFilter('')}

   237→                {completed.map(app => (
   238→                  <ApplicationCard
   239→                    key={app.id}
   240→                    app={app}
   241→                    timeAgo={timeAgo}

   250→
   251→      {/* Application detail dialog */}

   260→          </DialogHeader>
   261→          <div className="space-y-4">

   263→            {(() => {
   264→              const config = statusConfig[selectedApp.status] || { label: selectedApp.status, variant: 'secondary' as const, icon: Clock }
   265→              return <Badge variant={config.variant} className="w-fit">{config.label}</Badge>

   272→                  <Calendar className="h-3 w-3" /> Applied

   297→
   298→            {/* Progress timeline */}
   299→            <ApplicationTimeline status={selectedApp.status} appliedAt={selectedApp.applied_at} updatedAt={selectedApp.updated_at} />
   300→
   301→            {/* Cover letter */}

   304→                <h4 className="font-medium text-sm mb-1">Your Cover Letter</h4>

   318→                  <ExternalLink className="h-4 w-4" /> View Job

   323→                  variant="outline"
   324→                  onClick={() => {
   325→                    setSelectedApp(null)
   326→                    setWithdrawTarget(selectedApp)
   327→                  }}
   328→                  className="gap-2 text-destructive hover:text-destructive"

   338→      {/* Withdraw confirmation dialog */}

   342→            <DialogTitle className="flex items-center gap-2">
   343→              <AlertTriangle className="h-5 w-5 text-amber-500" />
   344→              Withdraw Application?
   345→            </DialogTitle>
   346→          </DialogHeader>

   351→              <strong>{withdrawTarget.company || withdrawTarget.posted_by_company || 'this company'}</strong>?
   352→            </p>
   353→            <div className="rounded-lg bg-amber-50 border border-amber-200 p-3 text-sm text-amber-800">
   354→              This action cannot be undone. You may need to reapply if you change your mind.
   355→            </div>
   356→            <div className="flex gap-2 pt-2">
   357→              <Button
   358→                variant="outline"
   359→                onClick={() => setWithdrawTarget(null)}

   366→                variant="destructive"

   378→            </div>

   382→    </div>

   384→}

   386→function ScreeningAnswersSection({
   387→  app,
   388→  parseScreeningData,
   389→}: {
   390→  app: Application
   391→  parseScreeningData: (app: Application) => { answers: Record<string, string> | null; questions: ScreeningQuestion[] }
   392→}) {
   393→  const { answers, questions } = parseScreeningData(app)
   394→
   395→  if (!answers || Object.keys(answers).length === 0) return null
   396→
   397→  return (
   398→    <div>
   399→      <h4 className="font-medium text-sm mb-2 flex items-center gap-1.5">
   400→        <ClipboardList className="h-4 w-4 text-primary" />
   401→        Your Screening Answers
   402→      </h4>
   403→      <div className="space-y-2">
   404→        {Object.entries(answers).map(([key, value], i) => {
   405→          const q = questions[i]
   406→          const questionText = q?.question || q || `Question ${i + 1}`
   407→          return (
   408→            <div key={key} className="rounded-lg bg-muted/50 p-3">
   409→              <p className="text-xs text-muted-foreground mb-1 font-medium">
   410→                {String(questionText)}
   411→              </p>
   412→              <p className="text-sm">{String(value)}</p>
   413→            </div>
   414→          )
   415→        })}
   416→      </div>
   417→    </div>
   418→  )
   419→}

   421→function ApplicationTimeline({ status, appliedAt, updatedAt }: { status: string; appliedAt: string; updatedAt: string }) {
   422→  const steps = [
   423→    { key: 'applied', label: 'Applied', icon: FileText },
   424→    { key: 'screening', label: 'Screening', icon: Eye },
   425→    { key: 'interviewed', label: 'Interview', icon: Briefcase },
   426→    { key: 'offered', label: 'Offer', icon: CheckCircle },
   427→  ]
   428→
   429→  const stepIndex = {
   430→    applied: 0, screening: 1, shortlisted: 1, reviewing: 1, interviewed: 2,
   431→    offered: 3, hired: 4, rejected: -1, withdrawn: -1,
   432→  }[status] ?? 0
   433→
   434→  // Special terminal states
   435→  if (status === 'rejected' || status === 'withdrawn') {
   436→    const config = statusConfig[status]
   437→    return (
   438→      <div>
   439→        <h4 className="font-medium text-sm mb-3">Status Timeline</h4>
   440→        <div className="space-y-3">
   441→          <div className="flex items-center gap-3">
   442→            <div className="flex h-8 w-8 items-center justify-center rounded-full border-2 border-primary bg-primary/10">
   443→              <FileText className="h-4 w-4 text-primary" />
   444→            </div>
   445→            <div>
   446→              <p className="text-sm font-medium">Applied</p>
   447→              <p className="text-xs text-muted-foreground">{new Date(appliedAt).toLocaleDateString()}</p>
   448→            </div>
   449→          </div>
   450→          <div className="ml-4 h-4 w-px bg-muted" />
   451→          <div className="flex items-center gap-3">
   452→            <div className={`flex h-8 w-8 items-center justify-center rounded-full border-2 ${
   453→              status === 'rejected' ? 'border-destructive bg-destructive/10' : 'border-muted bg-muted/50'
   454→            }`}>
   455→              <XCircle className={`h-4 w-4 ${status === 'rejected' ? 'text-destructive' : 'text-muted-foreground'}`} />
   456→            </div>
   457→            <div>
   458→              <p className="text-sm font-medium">{config.label}</p>
   459→              <p className="text-xs text-muted-foreground">{new Date(updatedAt).toLocaleDateString()}</p>
   460→            </div>
   461→          </div>
   462→        </div>
   463→      </div>
   464→    )
   465→  }
   466→
   467→  if (stepIndex < 0) return null
   468→
   469→  return (
   470→    <div>
   471→      <h4 className="font-medium text-sm mb-3">Progress</h4>
   472→      <div className="flex items-center gap-1">
   473→        {steps.map((step, i) => {
   474→          const isComplete = i <= stepIndex
   475→          const isCurrent = i === stepIndex
   476→          const Icon = step.icon
   477→          return (
   478→            <div key={step.key} className="flex items-center gap-1 flex-1">
   479→              <div className={`flex items-center gap-1.5 ${isComplete ? 'text-primary' : 'text-muted-foreground/50'}`}>
   480→                <div className={`flex h-7 w-7 items-center justify-center rounded-full border-2 transition-all ${
   481→                  isCurrent ? 'border-primary bg-primary text-white' :
   482→                  isComplete ? 'border-primary bg-primary/10' : 'border-muted'
   483→                }`}>
   484→                  <Icon className="h-3.5 w-3.5" />
   485→                </div>
   486→                <span className="text-[11px] font-medium hidden sm:inline">{step.label}</span>
   487→              </div>
   488→              {i < steps.length - 1 && (
   489→                <div className={`h-0.5 flex-1 mx-1 ${i < stepIndex ? 'bg-primary' : 'bg-muted'}`} />
   490→              )}
   491→            </div>
   492→          )
   493→        })}
   494→      </div>
   495→      {/* Date labels */}
   496→      <div className="flex justify-between mt-2 px-1">
   497→        <span className="text-[10px] text-muted-foreground">
   498→          {new Date(appliedAt).toLocaleDateString()}
   499→        </span>
   500→        {stepIndex > 0 && (
   501→          <span className="text-[10px] text-muted-foreground">
   502→            Updated {new Date(updatedAt).toLocaleDateString()}
   503→          </span>
   504→        )}
   505→      </div>
   506→    </div>
   507→  )
   508→}

   510→function ApplicationCard({
   511→  app,
   512→  timeAgo,
   513→  onWithdraw,
   514→  onClick,
   515→}: {
   516→  app: Application
   517→  timeAgo: (d: string) => string
   518→  onWithdraw?: () => void
   519→  onClick?: () => void
   520→}) {
   521→  const config = statusConfig[app.status] || { label: app.status, variant: 'secondary' as const, icon: Clock }
   522→
   523→  // Progress steps (aligned with pipeline stages)
   524→  const steps = ['Applied', 'Screening', 'Interview', 'Offer']
   525→  const currentStep = {
   526→    applied: 0, screening: 1, shortlisted: 1, reviewing: 1, interviewed: 2,
   527→    offered: 3, hired: 4, rejected: -1, withdrawn: -1,
   528→  }[app.status] ?? 0
   529→
   530→  // Check if has screening answers
   531→  let hasScreening = false
   532→  try {
   533→    const answers = typeof app.screening_answers === 'string' ? JSON.parse(app.screening_answers) : app.screening_answers
   534→    hasScreening = answers && Object.keys(answers).length > 0
   535→  } catch { /* ignore */ }
   536→
   537→  return (
   538→    <Card
   539→      className="cursor-pointer transition-shadow hover:shadow-md"
   540→      onClick={onClick}
