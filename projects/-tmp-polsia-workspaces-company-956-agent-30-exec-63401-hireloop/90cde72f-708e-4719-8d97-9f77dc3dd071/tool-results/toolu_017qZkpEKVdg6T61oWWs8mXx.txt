    65→function RoleRedirect() {
    66→  const { user, loading } = useAuth()
    67→
    68→  if (loading) return null
    69→  if (!user) return <Navigate to="/login" replace />
    70→  return <Navigate to={getDashboardPath(user.role)} replace />
    71→}
