   342→function ProfileCompleteness({ value }: { value: number }) {
   343→  const color = value >= 80 ? 'text-emerald-600' : value >= 50 ? 'text-amber-500' : 'text-red-500'
   344→  const bg = value >= 80 ? 'bg-emerald-600' : value >= 50 ? 'bg-amber-500' : 'bg-red-500'
   345→
   346→  return (
   347→    <div className="flex items-center gap-3">
   348→      <div className="text-right">
   349→        <p className={`text-sm font-bold ${color}`}>{value}%</p>
   350→        <p className="text-[10px] text-muted-foreground">Complete</p>
   351→      </div>
   352→      <div className="w-24 h-2 rounded-full bg-muted overflow-hidden">
   353→        <div className={`h-full rounded-full transition-all ${bg}`} style={{ width: `${value}%` }} />
   354→      </div>
   355→    </div>
   356→  )
   357→}

   359→// ============= Avatar Upload =============
   360→
   361→function AvatarUpload({ profile, onUploaded, showMessage }: {
   362→  profile: Profile
   363→  onUploaded: (url: string) => void
   364→  showMessage: (type: 'success' | 'error', text: string) => void
   365→}) {
   366→  const fileRef = useRef<HTMLInputElement>(null)
   367→  const [uploading, setUploading] = useState(false)
   368→
   369→  async function handleUpload(e: React.ChangeEvent<HTMLInputElement>) {
   370→    const file = e.target.files?.[0]
   371→    if (!file) return
   372→    setUploading(true)
   373→    try {
   374→      const formData = new FormData()
   375→      formData.append('photo', file)
   376→      const data = await apiCall<{ success: boolean; photo_url: string }>('/candidate/profile/photo', {
   377→        method: 'POST',
   378→        body: formData,
   379→        isFormData: true,
   380→      })
   381→      onUploaded(data.photo_url)
   382→      showMessage('success', 'Photo updated')
   383→    } catch {
   384→      showMessage('error', 'Failed to upload photo')
   385→    } finally {
   386→      setUploading(false)
   387→    }
   388→  }
   389→
   390→  return (
   391→    <div className="relative group">
   392→      <div className="h-20 w-20 rounded-full bg-muted flex items-center justify-center overflow-hidden border-2 border-border">
   393→        {profile.avatar_url ? (
   394→          <img src={profile.avatar_url} alt="Avatar" className="h-full w-full object-cover" />
   395→        ) : (
   396→          <User className="h-8 w-8 text-muted-foreground" />
   397→        )}
   398→      </div>
   399→      <button
   400→        onClick={() => fileRef.current?.click()}
   401→        disabled={uploading}
   402→        className="absolute inset-0 rounded-full bg-black/40 flex items-center justify-center opacity-0 group-hover:opacity-100 transition-opacity"
   403→      >
   404→        {uploading ? (
   405→          <div className="h-5 w-5 animate-spin rounded-full border-2 border-white border-t-transparent" />
   406→        ) : (
   407→          <Upload className="h-5 w-5 text-white" />
   408→        )}
   409→      </button>
   410→      <input ref={fileRef} type="file" accept="image/*" className="hidden" onChange={handleUpload} />
   411→    </div>
   412→  )
   413→}

  1050→function ResumeTab({ profile, setProfile, showMessage, onResumeApplied }: {
  1051→  profile: Profile
  1052→  setProfile: React.Dispatch<React.SetStateAction<Profile>>
  1053→  showMessage: (type: 'success' | 'error', text: string) => void
  1054→  onResumeApplied: () => Promise<void>

  1057→  const [uploading, setUploading] = useState(false)
  1058→  const [lastParsed, setLastParsed] = useState<ParsedData | null>(null)
  1059→  const [parseStep, setParseStep] = useState<string>('')
  1060→
  1061→  async function handleUpload(e: React.ChangeEvent<HTMLInputElement>) {
  1062→    const file = e.target.files?.[0]
  1063→    if (!file) return
  1064→    setUploading(true)
  1065→    setLastParsed(null)
  1066→    setParseStep('Uploading resume...')
  1067→    try {
  1068→      setParseStep('Analyzing with AI...')
  1069→      const formData = new FormData()
  1070→      formData.append('resume', file)
  1071→      const data = await apiCall<{ success: boolean; resume_url: string; parsed_data?: ParsedData }>('/candidate/resume/upload', {
  1072→        method: 'POST',
  1073→        body: formData,
  1074→        isFormData: true,
  1075→      })
  1076→      setProfile(p => ({ ...p, resume_url: data.resume_url }))
  1077→
  1078→      if (data.parsed_data) {
  1079→        setLastParsed(data.parsed_data)
  1080→        setParseStep('Refreshing profile...')

  1082→        await onResumeApplied()
  1083→
  1084→        const summary = data.parsed_data._applySummary
  1085→        if (summary) {
  1086→          const parts: string[] = []
  1087→          if (summary.profile) parts.push('profile info')
  1088→          if (summary.experience && summary.experience > 0) parts.push(`${summary.experience} job${summary.experience > 1 ? 's' : ''}`)
  1089→          if (summary.education && summary.education > 0) parts.push(`${summary.education} education`)
  1090→          if (summary.skills && summary.skills > 0) parts.push(`${summary.skills} skill${summary.skills > 1 ? 's' : ''}`)
  1091→
  1092→          if (parts.length > 0) {
  1093→            showMessage('success', `AI imported: ${parts.join(', ')}`)
  1094→          } else {
  1095→            showMessage('success', 'Resume parsed — profile already up to date')
  1096→          }
  1097→        } else {
  1098→          showMessage('success', 'Resume uploaded & parsed!')
  1099→        }
  1100→      } else {
  1101→        showMessage('success', 'Resume uploaded')
  1102→      }
  1103→      setParseStep('')
  1104→    } catch {
  1105→      showMessage('error', 'Failed to upload resume')
  1106→      setParseStep('')
  1107→    } finally {
  1108→      setUploading(false)
  1109→      // Reset file input so same file can be re-uploaded
  1110→      if (fileRef.current) fileRef.current.value = ''
  1111→    }
  1112→  }
  1113→

  1120→          {profile.resume_url ? (
  1121→            <div className="space-y-4">
  1122→              <div className="flex items-center gap-3 rounded-lg border p-4 bg-muted/50">
  1123→                <FileText className="h-8 w-8 text-primary shrink-0" />
  1124→                <div className="flex-1 min-w-0">
  1125→                  <p className="font-medium">Current Resume</p>
  1126→                  <p className="text-sm text-muted-foreground truncate">{profile.resume_url}</p>
  1127→                </div>
  1128→                <a href={profile.resume_url} target="_blank" rel="noopener noreferrer">
  1129→                  <Button variant="outline" size="sm">View</Button>
  1130→                </a>
  1131→              </div>
  1132→              <div className="flex gap-2">
  1133→                <Button onClick={() => fileRef.current?.click()} variant="outline" className="gap-1" disabled={uploading}>
  1134→                  {uploading ? (
  1135→                    <div className="h-4 w-4 animate-spin rounded-full border-2 border-primary border-t-transparent" />
  1136→                  ) : (
  1137→                    <Upload className="h-4 w-4" />
  1138→                  )}
  1139→                  {uploading ? 'Processing...' : 'Replace Resume'}
  1140→                </Button>

  1145→                  {parseStep}

  1151→              <div className="mx-auto mb-4 flex h-16 w-16 items-center justify-center rounded-full bg-muted">
  1152→                <Upload className="h-8 w-8 text-muted-foreground" />
  1153→              </div>
