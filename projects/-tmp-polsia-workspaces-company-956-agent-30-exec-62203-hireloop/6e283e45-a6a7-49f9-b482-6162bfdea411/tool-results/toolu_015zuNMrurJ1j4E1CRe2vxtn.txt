    21‚Üí// Helper function to extract text from various file formats
    22‚Üíasync function extractTextFromFile(buffer, mimetype) {
    23‚Üí  try {
    24‚Üí    // Handle PDF files
    25‚Üí    if (mimetype === 'application/pdf' || buffer.toString('utf-8', 0, 4) === '%PDF') {
    26‚Üí      const pdfData = await pdfParse(buffer);
    27‚Üí      return pdfData.text;
    28‚Üí    }
    29‚Üí
    30‚Üí    // Handle DOCX files (application/vnd.openxmlformats-officedocument.wordprocessingml.document)
    31‚Üí    if (mimetype === 'application/vnd.openxmlformats-officedocument.wordprocessingml.document' ||
    32‚Üí        buffer.toString('utf-8', 0, 2) === 'PK') {
    33‚Üí      const result = await mammoth.extractRawText({ buffer });
    34‚Üí      return result.value;
    35‚Üí    }
    36‚Üí
    37‚Üí    // Handle plain text files (fallback)
    38‚Üí    const text = buffer.toString('utf-8');
    39‚Üí    if (text.length > 0 && /[a-zA-Z]{3,}/.test(text.substring(0, 500))) {
    40‚Üí      return text;
    41‚Üí    }
    42‚Üí
    43‚Üí    throw new Error('Unsupported file format or unable to extract text');
    44‚Üí  } catch (error) {
    45‚Üí    console.error('Text extraction error:', error.message);
    46‚Üí    throw error;
    47‚Üí  }
    48‚Üí}
